Команди Gita

git config --global user.name blabla
git config --global user.email blabal@bla
git init -- створити порожнє сховище Git або повторно ініціалізувати існуюче
git status -- показати робочий статус дерева
git add -- додати вміст файлу до покажчика
git commit -- запис змін до репозиторію
git log -- показати журнали комітів
git checkout -- еревірка - перемикання гілок або відновлення робочих файлів дерева
git branch -- cписок, створення або видалення гілок
git push -- оновлення віддалених посилань разом із пов’язаними об’єктами
git pull -- отримати з іншого репозиторію або локальної гілки та інтегрувати з ним
git merge -- б'єднати дві або більше історій розробки разом
git clone -- клонувати репозиторій у новий каталог
.gitignore потрібний для приховання файлів та папок від системи контролю версій Git.
git revert -- повернути деякі наявні коміти
git reset -- cкидання поточного HEAD до вказаного стану
git rebase -- повторне застосування комітів поверх іншої базової підказки
git cherry-pick -- застосувати зміни, внесені деякими існуючими комітами
git stash -- заховати зміни в брудному робочому каталозі
